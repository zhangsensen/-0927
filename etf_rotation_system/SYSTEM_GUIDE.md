```python
                all_ics[:half],
                all_ices[half:2*half]

            # tæ£€éªŒ

                'ic_ir': ic_ir,

                'stability': stability,
                't_stat': t_stat,
                'p_value': p_value,
                'sample_size': len(all_ics),
                'coverage': len(factor_values) / len(factor_data)
            }
            result.update(period_ics)
            results.append(result)

    return pd.DataFrame(results)
```

```python
def apply_fdr_correction(ic_df, alpha=0.2):
    """
    Benjamini-Hochberg FDRæ ¡æ­£

    å‚æ•°:
    - ic_df: ICåˆ†æç»“æœDataFrame
    - alpha: æ˜¾è‘—æ€§æ°´å¹³

    è¿”å›:
    - é€šè¿‡FDRæ ¡æ­£çš„ç»“æœDataFrame

    return pd.DataFrame()  # æ— å› å­é€šè¿‡æ ¡æ­£
```

```python
def remove_correlated_factors(ic_df, panel, max_corr=0.7):
    """
    å»é™¤é«˜ç›¸å…³å› å­ï¼Œä¿ç•™IC_IRæ›´é«˜çš„å› å­

    å‚æ•°:
    - ic_df: ICåˆ†æç»“æœ
    - panel: å› å­é¢æ¿æ•°æ®
    - max_corr: æœ€å¤§ç›¸å…³æ€§é˜ˆå€¼

    è¿”å›:
    - å»é‡åçš„ICåˆ†æç»“æœ

        return ic_df

    factors = ic_df['factor'].tolist()
    factor_data = panel[factors]

    # è®¡ç®—ç›¸å…³æ€§çŸ©é˜µ

        method='spearman',
        min_periods=30

    for i, f1 in enumerate(factors):

            continue
        for f2 in factors[i+1:]:
            if f2 in to_remove:
                continue
            if corr_matrix.loc[f1, f2] > max_corr:
                # ä¿ç•™IC_IRæ›´é«˜çš„å› å­

                ir2 = ic_df[ic_df['factor'] == f2]['ic_ir'].values[0]

    return ic_df[~ic_df['factor'].isin(to_remove)].copy()
```
